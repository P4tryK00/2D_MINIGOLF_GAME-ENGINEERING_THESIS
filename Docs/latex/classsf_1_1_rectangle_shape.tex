\doxysubsection{Opis szczegółowy}
Specialized shape representing a rectangle. 

This class inherits all the functions of \doxylink{classsf_1_1_transformable}{sf\+::\+Transformable} (position, rotation, scale, bounds, ...) as well as the functions of \doxylink{classsf_1_1_shape}{sf\+::\+Shape} (outline, color, texture, ...).

Usage example\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_rectangle_shape}{sf::RectangleShape}}\ rectangle;}
\DoxyCodeLine{rectangle.\mbox{\hyperlink{classsf_1_1_rectangle_shape_a5c65d374d4a259dfdc24efdd24a5dbec}{setSize}}(\mbox{\hyperlink{classsf_1_1_vector2}{sf::Vector2f}}(100,\ 50));}
\DoxyCodeLine{rectangle.\mbox{\hyperlink{classsf_1_1_shape_a5978f41ee349ac3c52942996dcb184f7}{setOutlineColor}}(\mbox{\hyperlink{classsf_1_1_color_a127dbf55db9c07d0fa8f4bfcbb97594a}{sf::Color::Red}});}
\DoxyCodeLine{rectangle.\mbox{\hyperlink{classsf_1_1_shape_a5ad336ad74fc1f567fce3b7e44cf87dc}{setOutlineThickness}}(5);}
\DoxyCodeLine{rectangle.\mbox{\hyperlink{classsf_1_1_transformable_a4dbfb1a7c80688b0b4c477d706550208}{setPosition}}(10,\ 20);}
